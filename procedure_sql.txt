CREATE DEFINER=`root`@`localhost` PROCEDURE `insert_data_night`(
nuit INTEGER,
id_medecin_validateur INTEGER,
spo2_min DECIMAL(5,2),
spo2_moy DECIMAL(5,2),
spo2_mediane DECIMAL(5,2),
duree_sommeil_min INTEGER,
duree_hypoxie_min INTEGER,
position_dominante VARCHAR(20),
decibels_max DECIMAL(5,2),
decibels_moy DECIMAL(5,2),
nb_ronflements_forts INTEGER,
commentaire_medical TEXT)
BEGIN
	INSERT INTO resultat_nuit (id_nuit,id_medecin_validateur, date_validation, iah, spo2_min, spo2_moy, spo2_mediane, nb_apnees, nb_hypopnees, nb_rera, nb_microeveils, duree_sommeil_min, duree_hypoxie_min, position_dominante, duree_apnee_moy_sec, duree_apnee_max_sec, decibels_max, decibels_moy, nb_ronflements_forts,commentaire_medical)  VALUES
    (nuit,
    id_medecin_validateur, 
    current_date(),
    (SELECT SUM(valeur)*60/duree_sommeil_min FROM (select COUNT(type_evenement) as valeur FROM evenement_respiratoire WHERE (type_evenement = 'apnée obstructive' OR type_evenement = 'apnée centrale') AND id_nuit = nuit UNION ALL SELECT COUNT(type_evenement) as valeur FROM evenement_respiratoire WHERE (type_evenement = 'hypopnée') AND id_nuit = nuit) AS iah),
    spo2_min,
    spo2_moy,
    spo2_mediane,
    (SELECT (COUNT(type_evenement)) as nb_apnees FROM evenement_respiratoire WHERE (type_evenement = 'apnée obstructive' OR type_evenement = 'apnée centrale') AND id_nuit = nuit),
    (SELECT (COUNT(type_evenement)) as nb_hypopnees FROM evenement_respiratoire WHERE (type_evenement = 'hypopnée') AND id_nuit = nuit),
    (SELECT (COUNT(type_evenement)) as nb_rera FROM evenement_respiratoire WHERE (type_evenement = 'RERA') AND id_nuit = nuit),
    (SELECT (COUNT(type_evenement)) as nb_microeveils FROM evenement_respiratoire WHERE id_nuit = nuit),
    duree_sommeil_min,
    duree_hypoxie_min,
    position_dominante,
    (SELECT avg(duree_sec) as duree_apnee_moy_sec from evenement_respiratoire where id_nuit = nuit),
    (SELECT max(duree_sec) as duree_apnee_max_sec from evenement_respiratoire where id_nuit = nuit),
    decibels_max,
    decibels_moy,
    nb_ronflements_forts,
    commentaire_medical)
	ON DUPLICATE KEY UPDATE
    id_nuit = VALUES(id_nuit),
    id_medecin_validateur = VALUES(id_medecin_validateur),
    date_validation = current_date(),
    iah = VALUES(iah), 
    spo2_min= VALUES(spo2_min), 
    spo2_moy= VALUES(spo2_moy), 
    spo2_mediane= VALUES(spo2_mediane),
    nb_apnees= VALUES(nb_apnees),
    nb_hypopnees= VALUES(nb_hypopnees),
    nb_rera= VALUES(nb_rera),
    nb_microeveils= VALUES(nb_microeveils),
    duree_sommeil_min= VALUES(duree_sommeil_min),
    duree_hypoxie_min= VALUES(duree_hypoxie_min),
    position_dominante= VALUES(position_dominante), 
    duree_apnee_moy_sec= VALUES(duree_apnee_moy_sec),
    duree_apnee_max_sec= VALUES(duree_apnee_max_sec),
    decibels_max= VALUES(decibels_max),
    decibels_moy= VALUES(decibels_moy),
    nb_ronflements_forts= VALUES(nb_ronflements_forts),
    commentaire_medical=VALUES(commentaire_medical)
   ;
END
